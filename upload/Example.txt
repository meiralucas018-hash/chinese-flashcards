<script>
  (function () {
    var toast = document.getElementById('toast');
    var toastTimer = null;

    function showToast(msg) {
      if (!toast) return;
      toast.textContent = msg || 'Copied!';
      toast.classList.add('show');
      clearTimeout(toastTimer);
      toastTimer = setTimeout(function () {
        toast.classList.remove('show');
      }, 1400);
    }

    async function copyToClipboard(text) {
      try {
        if (navigator.clipboard && navigator.clipboard.writeText) {
          await navigator.clipboard.writeText(text);
          return true;
        }
      } catch (e) {}
      try {
        var ta = document.createElement('textarea');
        ta.value = text;
        ta.setAttribute('readonly', '');
        ta.style.position = 'fixed';
        ta.style.left = '-9999px';
        ta.style.top = '0';
        document.body.appendChild(ta);
        ta.focus();
        ta.select();
        var ok = document.execCommand('copy');
        document.body.removeChild(ta);
        return ok;
      } catch (e2) {
        return false;
      }
    }

    function wireMainCharCopy() {
      var mainChar = document.querySelector('.char-main');
      if (!mainChar) return;
      mainChar.addEventListener('click', async function (e) {
        e.preventDefault();
        e.stopPropagation();
        var txt = (this.textContent || '').trim();
        if (!txt) return;
        var ok = await copyToClipboard(txt);
        if (ok) showToast('Copied: ' + txt);
        else showToast('Copy failed');
      });
    }

    /* ═══════════════════════════════════════════════════════════════
       SNAP-TO-PERFECT DRAWING PAD with faint character template
       ───────────────────────────────────────────────────────────────
       Now includes a light-gray '对' in a beautiful calligraphic font
       (KaiTi) as a tracing guide. The template remains visible behind
       all strokes and is preserved across undo/clear.
    */
    function initDrawPad() {
      var canvas    = document.getElementById('drawingCanvas');
      var toggleBtn = document.getElementById('toggleDrawBtn');
      var clearBtn  = document.getElementById('clearDrawBtn');
      var undoBtn   = document.getElementById('undoDrawBtn');
      var container = document.getElementById('drawContainer');
      if (!canvas || !toggleBtn || !clearBtn || !undoBtn || !container) return;

      var ctx       = canvas.getContext('2d');
      var isDrawing = false;
      var rawPoints = [];

      // Draw a faint reference character in a refined Chinese font
      function drawTemplate() {
        ctx.save();
        // Elegant Kai / calligraphic font stack
        ctx.font = '180px "KaiTi", "楷体", "STKaiti", "华文楷体", "Kai", "Noto Serif SC", serif';
        ctx.fillStyle = 'rgba(255, 255, 255, 0.1)';  // very subtle
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText('对', canvas.width / 2, canvas.height / 2);
        ctx.restore();
      }

      // Undo stack of ImageData snapshots (index 0 = blank canvas with template)
      var undoStack = [];

      // ── Style for user strokes ─────────────────────────────────
      function applyStyle() {
        ctx.lineWidth   = 5;
        ctx.lineCap     = 'round';
        ctx.lineJoin    = 'round';
        ctx.strokeStyle = '#f1f5f9';
        ctx.shadowColor = 'rgba(96, 165, 250, 0.4)';
        ctx.shadowBlur  = 6;
      }
      applyStyle();

      // ── Coordinate helper ─────────────────────────────────────
      function getCoords(e) {
        var rect   = canvas.getBoundingClientRect();
        var scaleX = canvas.width  / rect.width;
        var scaleY = canvas.height / rect.height;
        var src    = (e.touches && e.touches.length) ? e.touches[0] : e;
        return {
          x: (src.clientX - rect.left) * scaleX,
          y: (src.clientY - rect.top)  * scaleY
        };
      }

      // ── Snapshot helpers ──────────────────────────────────────
      function restoreSnapshot(snap) { ctx.putImageData(snap, 0, 0); }
      function saveSnapshot()        { return ctx.getImageData(0, 0, canvas.width, canvas.height); }

      /* ── Chaikin smoothing (live preview only) ──────────────── */
      function chaikin(pts, passes) {
        var r = pts;
        for (var p = 0; p < (passes || 3); p++) {
          var n = [ r[0] ];
          for (var i = 0; i < r.length - 1; i++) {
            var a = r[i], b = r[i+1];
            n.push({ x: 0.75*a.x + 0.25*b.x, y: 0.75*a.y + 0.25*b.y });
            n.push({ x: 0.25*a.x + 0.75*b.x, y: 0.25*a.y + 0.75*b.y });
          }
          n.push(r[r.length - 1]);
          r = n;
        }
        return r;
      }

      function catmullSeg(p0, p1, p2, p3) {
        var cp1x = p1.x + (p2.x - p0.x) / 3;
        var cp1y = p1.y + (p2.y - p0.y) / 3;
        var cp2x = p2.x - (p3.x - p1.x) / 3;
        var cp2y = p2.y - (p3.y - p1.y) / 3;
        ctx.bezierCurveTo(cp1x, cp1y, cp2x, cp2y, p2.x, p2.y);
      }

      function drawLivePreview(pts) {
        if (pts.length < 2) return;
        var s   = chaikin(pts, 3);
        var g0  = { x: 2*s[0].x - s[1].x,               y: 2*s[0].y - s[1].y };
        var gN  = { x: 2*s[s.length-1].x - s[s.length-2].x, y: 2*s[s.length-1].y - s[s.length-2].y };
        var all = [g0].concat(s).concat([gN]);
        ctx.beginPath();
        ctx.moveTo(all[1].x, all[1].y);
        if (all.length <= 3) {
          ctx.lineTo(all[all.length-2].x, all[all.length-2].y);
        } else {
          for (var i = 1; i < all.length - 2; i++) {
            catmullSeg(all[i-1], all[i], all[i+1], all[i+2]);
          }
        }
        ctx.stroke();
      }

      var STRAIGHT_THRESHOLD = 0.10;   // 10% of chord → treated as straight

      function beautifyAndDraw(pts) {
        if (pts.length < 2) return;

        var p0 = pts[0];
        var pN = pts[pts.length - 1];

        // Arc-length parameterisation
        var arcLen = [0];
        for (var i = 1; i < pts.length; i++) {
          var ddx = pts[i].x - pts[i-1].x;
          var ddy = pts[i].y - pts[i-1].y;
          arcLen.push(arcLen[i-1] + Math.sqrt(ddx*ddx + ddy*ddy));
        }
        var totalLen = arcLen[arcLen.length - 1];
        if (totalLen < 2) {
          // Dot / tap
          ctx.beginPath();
          ctx.arc(p0.x, p0.y, 3, 0, Math.PI * 2);
          ctx.fillStyle = '#f1f5f9';
          ctx.fill();
          return;
        }
        var tArr = arcLen.map(function(l) { return l / totalLen; });

        // Straightness test
        var chordDx  = pN.x - p0.x;
        var chordDy  = pN.y - p0.y;
        var chordLen = Math.sqrt(chordDx*chordDx + chordDy*chordDy);
        var maxDev   = 0;

        for (var i = 1; i < pts.length - 1; i++) {
          var ex = pts[i].x - p0.x;
          var ey = pts[i].y - p0.y;
          var dev = (chordLen < 1)
            ? Math.sqrt(ex*ex + ey*ey)
            : Math.abs(ex * chordDy - ey * chordDx) / chordLen;
          if (dev > maxDev) maxDev = dev;
        }

        applyStyle(); // ensure style is still correct after putImageData

        if (chordLen < 1 || maxDev / chordLen < STRAIGHT_THRESHOLD) {
          /* ── PERFECT STRAIGHT LINE ── */
          ctx.beginPath();
          ctx.moveTo(p0.x, p0.y);
          ctx.lineTo(pN.x, pN.y);
          ctx.stroke();
          return;
        }

        // Least-squares quadratic Bézier fit
        var sumW2x = 0, sumW2y = 0, sumW2 = 0;

        for (var i = 1; i < pts.length - 1; i++) {
          var ti = tArr[i];
          var wi = 2 * ti * (1 - ti);
          if (wi < 1e-4) continue;

          var rx = pts[i].x - (1-ti)*(1-ti)*p0.x - ti*ti*pN.x;
          var ry = pts[i].y - (1-ti)*(1-ti)*p0.y - ti*ti*pN.y;

          sumW2x += wi * rx;
          sumW2y += wi * ry;
          sumW2  += wi * wi;
        }

        var P1;
        if (sumW2 < 1e-6) {
          P1 = { x: (p0.x + pN.x) / 2, y: (p0.y + pN.y) / 2 };
        } else {
          P1 = { x: sumW2x / sumW2, y: sumW2y / sumW2 };
        }

        /* ── PERFECT QUADRATIC BÉZIER CURVE ── */
        ctx.beginPath();
        ctx.moveTo(p0.x, p0.y);
        ctx.quadraticCurveTo(P1.x, P1.y, pN.x, pN.y);
        ctx.stroke();
      }

      // ── Event handlers ────────────────────────────────────────
      function onStart(e) {
        e.preventDefault();
        isDrawing = true;
        rawPoints = [];
        rawPoints.push(getCoords(e));
      }

      function onMove(e) {
        if (!isDrawing) return;
        e.preventDefault();
        var c    = getCoords(e);
        var last = rawPoints[rawPoints.length - 1];
        var dx   = c.x - last.x, dy = c.y - last.y;
        if (dx*dx + dy*dy < 4) return;  // ~2px distance gate
        rawPoints.push(c);

        // Live Chaikin preview (template is preserved because we restore snapshot)
        restoreSnapshot(undoStack[undoStack.length - 1]);
        applyStyle();
        drawLivePreview(rawPoints);
      }

      function onEnd() {
        if (!isDrawing) return;
        isDrawing = false;

        // Replace the rough preview with the perfect beautified stroke
        restoreSnapshot(undoStack[undoStack.length - 1]);
        beautifyAndDraw(rawPoints);

        // Commit to undo stack
        undoStack.push(saveSnapshot());
        rawPoints = [];
      }

      // Mouse
      canvas.addEventListener('mousedown',  onStart);
      canvas.addEventListener('mousemove',  onMove);
      canvas.addEventListener('mouseup',    onEnd);
      canvas.addEventListener('mouseleave', onEnd);

      // Touch
      canvas.addEventListener('touchstart',  onStart, { passive: false });
      canvas.addEventListener('touchmove',   onMove,  { passive: false });
      canvas.addEventListener('touchend',    onEnd);
      canvas.addEventListener('touchcancel', onEnd);

      // ── Undo ──────────────────────────────────────────────────
      undoBtn.addEventListener('click', function () {
        if (undoStack.length > 1) {
          undoStack.pop();
          restoreSnapshot(undoStack[undoStack.length - 1]);
        }
      });

      // ── Clear ─────────────────────────────────────────────────
      clearBtn.addEventListener('click', function () {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        drawTemplate();                      // re-draw the faint character
        undoStack = [ saveSnapshot() ];      // new clean state with template
        rawPoints = [];
      });

      // ── Toggle ────────────────────────────────────────────────
      toggleBtn.addEventListener('click', function () {
        if (container.classList.contains('hidden')) {
          container.classList.remove('hidden');
          this.textContent = 'Close Drawing Pad';
        } else {
          container.classList.add('hidden');
          this.textContent = 'Practice Drawing';
        }
      });

      // ── Initialise canvas with template and first snapshot ────
      drawTemplate();
      undoStack.push(saveSnapshot());
    }

    function wireCharCopy() {
      document.querySelectorAll('.char').forEach(function (ch) {
        ch.addEventListener('click', async function (e) {
          e.preventDefault();
          e.stopPropagation();
          var txt = (ch.textContent || '').trim();
          if (!txt) return;
          var ok = await copyToClipboard(txt);
          if (ok) showToast('Copied: ' + txt);
          else showToast('Copy failed');
        });
      });
    }

    function wireWordCopy() {
      document.querySelectorAll('.word').forEach(function (w) {
        w.addEventListener('click', async function (e) {
          if (e.target && e.target.closest && e.target.closest('.char')) return;
          e.preventDefault();
          e.stopPropagation();
          var txt = (w.textContent || '').replace(/\s+/g, '').trim();
          if (!txt) return;
          var ok = await copyToClipboard(txt);
          if (ok) showToast('Copied: ' + txt);
          else showToast('Copy failed');
        });
      });
    }

    function wireSentenceCopy() {
      document.querySelectorAll('.sentence-wrap[data-clickcopy="true"]').forEach(function (wrap) {
        wrap.addEventListener('click', async function (e) {
          if (e.target.closest('.word') || e.target.closest('.char')) return;
          var text = wrap.getAttribute('data-copy') || '';
          if (!text) return;
          var ok = await copyToClipboard(text);
          if (ok) showToast('Copied: ' + text);
          else showToast('Copy failed');
        });
      });
    }

    function wireTooltipSuppress() {
      document.querySelectorAll('.word .char').forEach(function (ch) {
        ch.addEventListener('mouseenter', function () {
          var w = ch.closest('.word');
          if (w) w.classList.add('suppress-word-tooltip');
        });
        ch.addEventListener('mouseleave', function () {
          var w = ch.closest('.word');
          if (w) w.classList.remove('suppress-word-tooltip');
        });
        ch.addEventListener('touchstart', function () {
          var w = ch.closest('.word');
          if (w) w.classList.add('suppress-word-tooltip');
          setTimeout(function () {
            if (w) w.classList.remove('suppress-word-tooltip');
          }, 800);
        }, { passive: true });
      });
    }

    wireMainCharCopy();
    initDrawPad();
    wireCharCopy();
    wireWordCopy();
    wireSentenceCopy();
    wireTooltipSuppress();
  })();
